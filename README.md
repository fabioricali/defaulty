## Installation

### Node.js
```
npm install defaulty --save
```

## Example
```javascript
const defaulty = require('defaulty');

let defaultObj = {a: 1, b: 2, c: 3, d: {a: 5, b: 2}};
let targetObj = {a: 4, b: 5, d: {a: 1}};

let result = defaulty(targetObj, defaultObj);

console.log(result); //=> {a: 4, b: 5, c: 3, d: {a: 1, b: 2}};
```
